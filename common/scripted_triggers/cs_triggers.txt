
has_no_anticrime_active = {
	OR = {
		NOT = { has_modifier = anticrime_campaign }
		AND = {
			num_assigned_jobs = {
				job = enforcer
				value = 0
			}
			num_assigned_jobs = {
				job = telepath
				value = 0
			}
		}
	}
}

enforcers_amount = {
	OR = {
		num_assigned_jobs = {
			job = enforcer
			value >= $AMOUNT$
		}
		num_assigned_jobs = {
			job = telepath
			value >= $AMOUNT$
		}
	}
}

has_same_sector_leader = {
	exists = sector
		sector = {
			exists = leader
			leader = {
				is_same_value = event_target:dismissed_leader
			}
		}
}

has_corrupted_trait = {
	exists = sector
		sector = {
			exists = leader
			leader = {
				has_trait = leader_trait_corrupt
			}
		}
}


has_unemployed_amount_demand = {
	count_owned_pop = {
		limit = {
			pop_has_happiness = yes
			#is_unemployed = yes
			has_job = unemployed
			is_sapient = yes
			NOR = {
				has_living_standard = {
					country = fromfrom
					type = living_standard_chemical_bliss
				}
				has_citizenship_type = {
					country = fromfrom
					type = citizenship_assimilation
				}
			}
		}
		count >= $NUMBER$
	}
}

count_pop_faction = {
	count_owned_pops = {
		limit = {
			exists = pop_faction
			pop_faction = {
				is_pop_faction_type = $FACTION_TYPE$
			}
		}
		count >= $NUMBER$
	}
}


count_pop_of_faction = {
	count_owned_pops = {
		limit = {
			exists = pop_faction
			pop_faction = {
				is_same_value = event_target:cs_active_faction
			}
		}
		count >= $NUMBER$
	}
}


count_faction_on_planet = {
	event_target:cs_active_faction = {
		is_pop_faction_type = $FACTION_TYPE$
		root = {
			count_owned_pops = {
				limit = {
					exists = pop_faction
					pop_faction = {
						is_pop_faction_type = $TARGET_FACTION_TYPE$
					}
				}
				count >= $NUMBER$
			}
		}
	}
}


count_aliens_on_planet = {
	event_target:cs_active_faction = {
		is_pop_faction_type = isolationist
		root = {
			count_owned_pops = {
				limit = {
					NOT = { is_same_species = owner_main_species }
					#is_majority_species = this.species
				}
				count >= $NUMBER$
			}
		}
	}
}

	OR = {
		NOT = {
			count_faction_on_planet = {
				FACTION_TYPE = technologist
				TARGET_FACTION_TYPE = traditionalist
				NUMBER = $NUMBER$
			}
		}
		NOT = {
			count_faction_on_planet = {
				FACTION_TYPE = traditionalist
				TARGET_FACTION_TYPE = technologist
				NUMBER = $NUMBER$
			}
		}
		NOT = {
			count_faction_on_planet = {
				FACTION_TYPE = xenoist
				TARGET_FACTION_TYPE = supremacist
				NUMBER = $NUMBER$
			}
		}
		NOT = {
			count_faction_on_planet = {
				FACTION_TYPE = supremacist
				TARGET_FACTION_TYPE = xenoist
				NUMBER = $NUMBER$
			}
		}
		NOT = {
			count_faction_on_planet = {
				FACTION_TYPE = imperialist
				TARGET_FACTION_TYPE = prosperity
				NUMBER = $NUMBER$
			}
		}
		NOT = {

				FACTION_TYPE = prosperity

				NUMBER = $NUMBER$

pop_cant_flee = {
	OR = {
		is_sapient = no
		is_enslaved = yes
		pop_has_happiness = no
		has_citizenship_type = { 
			country = fromfrom
			type = citizenship_assimilation 
		}
		has_citizenship_type = { 
			country = fromfrom
			type = citizenship_slavery
		}
		has_living_standard = {
			country = fromfrom
			type = living_standard_chemical_bliss
		}
	}
}

	OR = {
		AND = {
			#former_citizenship_type = citizenship_limited
			root = {
				check_variable = {
					which = citizenship_lvl
					value = 1
				}
			}
			has_citizenship_type = {
				country = event_target:cs_target_country
				type = citizenship_full
			}
		}
		AND = {
			#former_living_standard_type = living_standard_normal
			root = {
				check_variable = {
					which = living_standart_lvl
					value = 2
				}
			}
			OR = {
				has_living_standard = {
					country = event_target:cs_target_country
					type = living_standard_good
				}

					country = event_target:cs_target_country
					type = living_standard_utopian
				}

					country = event_target:cs_target_country
					type = living_standard_academic_privilege
				}

					country = event_target:cs_target_country
					type = living_standard_shared_burden
				}

		}

				}
				has_living_standard = {
					country = event_target:cs_target_country

				}

				}

cs_is_species_count = {
	root = {
		count_pops = {
			limit = {
				#can_have_better_rights = yes
				is_exact_same_species = prev
			}
			count >= $NUMBER$
		}
	}
}

		is_same_value = root
	}
	any_system_planet = {
		is_planet_class = $STAR_TYPE$
	}
}

		}
	}
