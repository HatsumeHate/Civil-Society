create_martyr_leader = {
	owner = {
		random_list = {
			1 = {
				create_leader = {
					class = governor
					species = owner_main_species
					name = random
					skill = 3
					traits = {
						trait = leader_trait_governor_psionic
						trait = random_trait
					}
				}
			}
			1 = {
				create_leader = {
					class = scientist
					species = owner_main_species
					name = random
					skill = 3
					traits = {
						trait = leader_trait_scientist_psionic
						trait = random_trait
					}
				}
			}
			1 = {
				create_leader = {
					class = admiral
					species = owner_main_species
					name = random
					skill = 3
					traits = {
						trait = leader_trait_admiral_psionic
						trait = random_trait
					}
				}
			}
		}
		last_created_leader = {
			set_age = 25
			set_leader_flag = martyr_leader_flag
		}
	}
}

setup_rights = {
	if = {
		limit = {
			has_citizenship_type = {
				country = fromfrom
				type = citizenship_limited
			}
		}
		root = {
			set_variable = {
				which = citizenship_lvl
				value = 1
			}
		}
	}
	else_if = {
		limit = {
